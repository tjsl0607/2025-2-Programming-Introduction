int main(void)
{
	float sunny_speed = 300000.0; 
	double SunEarthInstance = 14960*10000;
	double time;

	time = SunEarthInstance/sunny_speed;

	printf("빛의 속도는 %12.6fkm/s",sunny_speed);
	printf("태양과 지구와의 거리 %.6lf", SunEarthInstance);
	printf("도달시간은 %.6lf초", time);
	
	return 0;
}
